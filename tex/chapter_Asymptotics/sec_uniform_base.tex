\subsection{–í–∏–≤–µ–¥–µ–Ω–Ω—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–æ–≥–æ —Ä—ñ–≤–Ω—è–Ω–Ω—è}

–ó—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ –ø–æ—Ä—è–¥–æ–∫ –≤–∏–±–æ—Ä—É –≤—ñ–ª—å–Ω–∏—Ö –ø—Ä–æ–º—ñ–∂–∫—ñ–≤ –≤–æ–¥—ñ—è–º–∏ –Ω–µ –≤–ø–ª–∏–≤–∞—î –Ω–∞ —Ä–µ–∑—É–ª—å—Ç–∞—Ç, —Ç–æ–º—É –±—É–¥–µ–º–æ –≤–≤–∞–∂–∞—Ç–∏, —â–æ –ø—ñ—Å–ª—è –ø–∞—Ä–∫—É–≤–∞–Ω–Ω—è –æ–¥–Ω–æ–≥–æ –∞–≤—Ç–æ–º–æ–±—ñ–ª—è –ø–∞—Ä–∫–æ–≤–∫–∞ —Ä–æ–∑–±–∏–≤–∞—î—Ç—å—Å—è –Ω–∞ 2 —á–∞—Å—Ç–∏–Ω–∏, —ñ –ø—ñ—Å–ª—è —Ü—å–æ–≥–æ —Å–ø–æ—á–∞—Ç–∫—É –∑–∞–ø–æ–≤–Ω—é—î—Ç—å—Å—è –ª—ñ–≤–∞ —á–∞—Å—Ç–∏–Ω–∞, –∞ –ø–æ—Ç—ñ–º –ø—Ä–∞–≤–∞.

–ù–µ–æ–±—Ö—ñ–¥–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ $m(X)=\EE F(X)$. –ù–µ—Ö–∞–π $\xi \sim Uniform(0, X - 1)$ ‚Äì –≤–∏–ø–∞–¥–∫–æ–≤–∞ –≤–µ–ª–∏—á–∏–Ω–∞, —â–æ –≤–∏–∑–Ω–∞—á–∞—î –ø–æ–ª–æ–∂–µ–Ω–Ω—è –ª—ñ–≤–æ–≥–æ –∫—Ä–∞—é –ø–µ—Ä—à–æ–≥–æ –∞–≤—Ç–æ–º–æ–±—ñ–ª—è –Ω–∞ –ø–∞—Ä–∫–æ–≤—Ü—ñ. –¢–æ–¥—ñ –º–∞—î–º–æ –Ω–∞—Å—Ç—É–ø–Ω—É —Ç–æ—Ç–æ–∂–Ω—ñ—Å—Ç—å:
\[
\begin{split}
	m(X) = \EE(\EE(F(\xi) + F(X - 1 - \xi) + 1 | \xi)) =\\
	=\int\limits_0^{X-1} ùëö(ùë°) \frac{1}{ùëã ‚àí 1} ùëëùë° + \int\limits_0^{X-1} ùëö(ùëã ‚àí ùë° ‚àí 1) \frac{1}{ùëã ‚àí 1} ùëëùë° + 1
\end{split}
\]

–¢–∞–∫ —è–∫
$$\int\limits_0^{X-1} ùëö(ùëã ‚àí ùë° ‚àí 1) ùëëùë° = \< u=X - t - 1 \> = - \int\limits_{X-1}^0 ùëö(u) ùëëùë° = \int\limits_0^{X-1} ùëö(u)ùëëùë°,$$

—Ç–æ
\begin{equation}
	m(X) = \frac{2}{X-1} \int\limits_0^{X-1} ùëö(ùë°) dt + 1
\end{equation}

–î–ª—è –∑—Ä—É—á–Ω–æ—Å—Ç—ñ –∑—Ä–æ–±–∏–º–æ –∑–∞–º—ñ–Ω—É $X \rightarrow X + 1$. –û—Ç—Ä–∏–º–∞—î–º–æ

\begin{equation}
	\label{eq:uniform_equation}
	m(X + 1) = \frac{2}{X} \int\limits_0^{X} ùëö(ùë°) dt + 1,\quad \forall X > 0
\end{equation}

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –æ—Ç—Ä–∏–º–∞–ª–∏ —ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è. –î–æ —Ç–æ–≥–æ –∂, –≤—ñ–¥–æ–º–æ, —â–æ
\begin{equation}
	\label{eq:zero_state}
	m(X) \equiv 0, \quad X \in [0; 1)
\end{equation}

–°–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ \eqref{eq:upperbound} —Ç–∞ \eqref{eq:lowerbound}, –º–∞—î–º–æ –æ–±–º–µ–∂–µ–Ω–Ω—è –Ω–∞ $m(X)$:
\begin{equation}
	\label{eq:bounds}
	\left[\frac{X+1}{2}\right] \leq m(X) \leq \left[X\right]
\end{equation}

–ó —Ü—ñ—î—ó –Ω–µ—Ä—ñ–≤–Ω–æ—Å—Ç—ñ –≤–∏–ø–ª–∏–≤–∞—î, —â–æ —è–∫—â–æ —î —è–∫–∞—Å—å –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫–∞ —É —Ñ—É–Ω–∫—Ü—ñ—ó $m(X)$, —Ç–æ –≤–æ–Ω–∞ –ø–æ—Ä—è–¥–∫—É $X$, —Ç–æ–±—Ç–æ
\begin{equation}
	\label{eq:bounds_inference}
	m(X) \sim C\cdot X \text{ –ø—Ä–∏ } x \rightarrow +\infty, \quad C \in [0.5; 1]
\end{equation}

\subsection{–ü–µ—Ä–µ—Ö—ñ–¥ –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞}
\label{sec:uniform_integral_laplace}

–°–ø—Ä–æ–±—É—î–º–æ —Ä–æ–∑–≤'—è–∑–∞—Ç–∏ \eqref{eq:uniform_equation} –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –ø–µ—Ä–µ—Ç–≤–æ—Ä–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞.

–û—Å–∫—ñ–ª—å–∫–∏ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è \eqref{eq:bounds}, —Ç–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –¥–ª—è $m(X)$ —ñ—Å–Ω—É—î. –î–æ —Ç–æ–≥–æ –∂, –∑–∞ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—é \eqref{eq:laplace_lag}:
$$\Lapl{m(X+1)}=\<\eqref{eq:zero_state}\>=\Lapl{m(X+1)\eta(X+1)}=e^p M(p).$$

–û—Å–∫—ñ–ª—å–∫–∏ $m(X) \leq X$, —Ç–æ $\int\limits_0^X m(t) dt < X^2$, —Ç–æ–±—Ç–æ –¥–ª—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª—É –≤—ñ–¥ $m(X)$ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è —Ç–∞–∫–æ–∂ —ñ—Å–Ω—É—î, –∑–∞ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—é \eqref{eq:laplace_int_origin}:
$$\Lapl{\int\limits_0^X m(t) dt} = \frac{M(p)}{p}.$$

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥–æ–≤–æ–¥–∏—Ç—å—Å—è, —â–æ $\frac{1}{X}\int\limits_0^X m(t) dt < X$ –ø—Ä–∏ $X>0$, –∞ —Ç–æ–º—É –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –¥–ª—è —Ü—å–æ–≥–æ –≤–∏—Ä–∞–∑—É —Ç–∞–∫–æ–∂ —ñ—Å–Ω—É—î. –¢–æ–¥—ñ –∑–∞ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—é \eqref{eq:laplace_int_image}:
$$\Lapl{\frac{2}{X}\int\limits_0^X m(t) dt} = 2 \int\limits_p^\infty \frac{M(s)}{s} ds.$$

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –æ—Ç—Ä–∏–º–∞–ª–∏ —ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è –≤ —Ç–µ—Ä–º—ñ–Ω–∞—Ö –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞, —è–∫–µ –≤–∂–µ –º–æ–∂–Ω–∞ —Ä–æ–∑–≤'—è–∑–∞—Ç–∏, –∞–¥–∂–µ –Ω–µ–º–∞ –∑—Å—É–≤—É:
\begin{equation}
	\label{eq:uniform_laplace_integral}
	e^p M(p) = 2 \int\limits_p^\infty \frac{M(s)}{s} ds + \frac{1}{p}
\end{equation}

–ü—Ä–æ–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é—î–º–æ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏ —Ä—ñ–≤–Ω—è–Ω–Ω—è –∑–∞ $p$:
\begin{equation}
	e^p M(p) + e^p \dot M(p) = - 2\frac{M(p)}{p}  - \frac1{p^2}
\end{equation}

–í–∏—Ä–∞–∑–∏–º–æ $\dot M(p)$ –∑ —Ü—å–æ–≥–æ —Ä—ñ–≤–Ω—è–Ω–Ω—è:
\begin{equation}
	\label{eq:uniform_laplace_diff}
	\dot M(p) = -M(p) - \frac{2M(p)}{pe^p} - \frac{1}{p^2 e^p} = - M(p)\left(1 + \frac{2e^{-p}}{p}\right) - \frac{e^{-p}}{p^2}
\end{equation}

–†–æ–∑–≤'—è–∂–µ–º–æ –æ—Ç—Ä–∏–º–∞–Ω–µ –¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ–π–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è. –°–ø–æ—á–∞—Ç–∫—É —Ä–æ–∑–≤'—è–∂–µ–º–æ –æ–¥–Ω–æ—Ä—ñ–¥–Ω—É —á–∞—Å—Ç–∏–Ω—É:
\begin{align*}
	\dot M_h (p) &= - M_h (p)\left(1 + \frac{2e^{-p}}{p}\right) \\
	\frac{\dot M_h (p)}{M_h (p)} &= -\left(1 + \frac{2e^{-p}}{p}\right) \\
	\int\limits_1^p \frac{\dot M_h (s)}{M_h (s)} ds &= -\int\limits_1^p \left(1 + \frac{2e^{-s}}{s}\right) ds \\
	\left. \ln{M_h (s)}\right|_1^p &= - (p - 1) - 2 \int\limits_1^p \frac{e^{-s}}{s} ds
\end{align*}

–ü–æ–∑–Ω–∞—á–∏–º–æ
\begin{equation}
	\label{eq:almost_li}
	Q(p) := \int\limits_1^p \frac{e^{-s}}{s} ds
\end{equation}

–¢–æ–¥—ñ –º–∞—î–º–æ
\begin{align*}
	\ln{M_h (p)} &= \ln{M_h (1)} + 1 - p - 2Q(p) \\
	M_h (p) &= M_h (1) \cdot e^{1 - p - 2Q(p)} \cdot const
\end{align*}

–û—Å–∫—ñ–ª—å–∫–∏ $M(1)$ –º–æ–∂–Ω–∞ –≤–∫–ª—é—á–∏—Ç–∏ –≤ –∫–æ–Ω—Å—Ç–∞–Ω—Ç—É, —Ç–æ –º–∞—î–º–æ —Ä–æ–∑–≤'—è–∑–æ–∫
\begin{equation}
	\label{eq:uniform_laplace_homogen_sol}
	M_h (p) = C \cdot e^{1 - p - 2Q(p)}, \quad \forall C \in \RR
\end{equation}

–î—ñ–π—Å–Ω–æ, –ø–µ—Ä–µ–≤—ñ—Ä–∏–º–æ —Ü–µ–π —Ä–æ–∑–≤'—è–∑–æ–∫:
\[
\begin{split}
	&\dot M_h (p) = C \cdot e^{1 - p - 2Q(p)} \cdot (-1 - 2 \dot Q(p)) =\\
	 &=C \cdot e^{1 - p - 2Q(p)} \cdot \left(-1 - 2\frac{e^{-p}}{p}\right)=
	- M_h (p) \cdot \left(1 + 2\frac{e^{-p}}{p}\right)
\end{split}
\]

–ù–µ—Å–∫–ª–∞–¥–Ω–æ –ø–æ–º—ñ—Ç–∏—Ç–∏, —â–æ –æ—Ç—Ä–∏–º–∞–ª–∏ –≤–∏—Ö—ñ–¥–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è. –¢–µ–ø–µ—Ä –∑–∞—Å—Ç–æ—Å—É—î–º–æ –º–µ—Ç–æ–¥ –≤–∞—Ä—ñ–∞—Ü—ñ—ó –¥–æ–≤—ñ–ª—å–Ω–∏—Ö —Å—Ç–∞–ª–∏—Ö:
\begin{align*}
	M(p) &= C(p) \cdot e^{1 - p - 2Q(p)} \\
	\dot M(p) &= \dot C(p) \cdot e^{1 - p - 2Q(p)} - C(p) \cdot (e^{1 - p - 2Q(p)}) \cdot \left(1 + 2 \frac{e^{-p}}{p}\right) \\
	&\text{–∑ —ñ–Ω—à–æ—ó —Å—Ç–æ—Ä–æ–Ω–∏, –∑ \eqref{eq:uniform_laplace_diff} –º–∞—î–º–æ} \\
	 \dot M(p) &= - C(p) \cdot e^{1 - p - 2Q(p)} \left(1 + 2 \frac{e^{-p}}{p}\right) - \frac{e^{-p}}{p^2} \quad	\\
	 &\text{—Ç–æ–º—É} \\
	 \dot C(p) &= -\frac{e^{-p}}{p^2} e^{p+2Q(p) - 1} = -e^{-1}\frac{e^{2Q(p)}}{p^2} \\
\end{align*}

–¢–æ–¥—ñ –ø—Ä–æ—Å—Ç–∏–º —ñ–Ω—Ç–µ–≥—Ä—É–≤–∞–Ω–Ω—è–º –≤ –º–µ–∂–∞—Ö –≤—ñ–¥ 1 –¥–æ $p$ –æ—Ç—Ä–∏–º—É—î–º–æ:
\begin{equation}
	C(p) = - \exp(-1) \cdot \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + const
\end{equation}

–Ü —Ç–æ–¥—ñ –æ—Ç—Ä–∏–º—É—î–º–æ –≤–∏—Ä–∞–∑ –¥–ª—è $M(p)$:
\begin{equation}
\begin{split}
	M(p)=- \left( \exp(-1) \cdot \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + const \right) e^{1 - p - 2Q(p)} =\\
	= - \left( \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + K \right) e^{- p - 2Q(p)}, \quad K \in \RR
\end{split}
\end{equation}

–ü–µ—Ä–µ–≤—ñ—Ä–∏–º–æ –æ—Ç—Ä–∏–º–∞–Ω–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:
\begin{equation*}
\begin{split}
	&\dot M(p) = -\left( \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + K \right)' e^{- p - 2Q(p)} -\left( \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + K \right)\cdot \\ 
	&\cdot \left(e^{- p - 2Q(p)}\right)' = -\frac{e^{2Q(p)}}{p^2} e^{- p - 2Q(p)}  + \left( \int\limits_1^p \frac{e^{2Q(s)}}{s^2} ds + K \right) \cdot \\
	& \cdot e^{- p - 2Q(p)} \left(1 + 2 \frac{e^{-p}}{p}\right) = -\frac{e^{-p}}{p^2} - M(p)\left(1 + 2 \frac{e^{-p}}{p}\right)
\end{split}
\end{equation*}

–ü–µ—Ä–µ–≤—ñ—Ä–µ–Ω–æ. –¢–æ–¥—ñ –æ—Å—Ç–∞—Ç–æ—á–Ω–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç –±–µ–∑ –≤–∏—Ä–∞—Ö—É–≤–∞–Ω–Ω—è –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏:
\begin{equation}
	\label{eq:uniform_laplace_sol_initial}
	M(p)= \left( \int\limits_p^1 \frac{e^{2Q(s)}}{s^2} ds + K \right) e^{- p - 2Q(p)}
\end{equation}

\subsection{–í–∏–∑–Ω–∞—á–µ–Ω–Ω—è –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ —É —Ä–æ–∑–≤'—è–∑–∫—É}

–£ \ref{sec:uniform_integral_laplace} –±—É–ª–æ –¥–æ–≤–µ–¥–µ–Ω–æ, —â–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ —ñ—Å–Ω—É—î –Ω–µ —Ç—ñ–ª—å–∫–∏ –¥–ª—è $m(X)$, –∞ —ñ –¥–ª—è $m(X+1)$, –¥–æ —Ç–æ–≥–æ –∂,
\begin{equation}
	\Lapl{m(X+1)} = e^p M(p).
\end{equation}

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –º–∞—î–º–æ
\begin{equation}
	\Lapl{m(X+1)} =\tilde M(p)= \left( \int\limits_p^1 \frac{e^{2Q(s)}}{s^2} ds + K \right) e^{- 2Q(p)}.
\end{equation}

–û—Å–∫—ñ–ª—å–∫–∏ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ ‚Äì –∞–Ω–∞–ª—ñ—Ç–∏—á–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –≤ –¥–µ—è–∫—ñ–π –ø—Ä–∞–≤—ñ–π –ø—ñ–≤–ø–ª–æ—â–∏–Ω—ñ –∫–æ–º–ø–ª–µ–∫—Å–Ω–æ–≥–æ –ø—Ä–æ—Å—Ç–æ—Ä—É, —Ç–æ $\tilde M(p) \rightarrow 0,\; p \rightarrow +\infty$.

–†–æ–∑–≥–ª—è–Ω–µ–º–æ $Q(p)$ ($p$ —Ä–æ–∑–≥–ª—è–¥–∞—î–º–æ –Ω–∞ –¥—ñ–π—Å–Ω—ñ–π –≤—ñ—Å—ñ):
\begin{equation}
\begin{split}
	&Q(p) = \int\limits_1^p \frac{e^{-s}}{s} ds <  \int\limits_1^\infty \frac{e^{-s}}{s} ds < 
	\int\limits_1^\infty e^{-s} ds=\\
	&= \exp(-1) - \exp(-\infty) = \exp(-1)
\end{split}
\end{equation}

–¢–æ–±—Ç–æ $Q(p)$ - –æ–±–º–µ–∂–µ–Ω–∞ –Ω–∞ $[1; \infty]$. –¢–æ–º—É –æ–±–º–µ–∂–µ–Ω–∏–º–∏ –Ω–∞ —Ü—ñ–π –≤—ñ—Å—ñ –±—É–¥—É—Ç—å —ñ $e^{\pm 2Q(p)}$. –¢–∞–∫–æ–∂ –∑—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ —è–∫—â–æ —ñ–Ω—Ç–µ–≥—Ä—É–≤–∞—Ç–∏ –ø–æ –¥—ñ–π—Å–Ω—ñ–π –≤—ñ—Å—ñ, —Ç–æ $Q(p)$ ‚Äì –º–æ–Ω–æ—Ç–æ–Ω–Ω–æ –∑—Ä–æ—Å—Ç–∞—é—á–∞ –∑–∞ $p$. –¢–æ–º—É
\begin{equation}
	0 = \tilde M(\infty) = \lim_{p\rightarrow \infty} \tilde M(p) = \left( \int\limits_\infty^1 \frac{e^{2Q(s)}}{s^2} ds + K \right) \lim_{p\rightarrow \infty} e^{- 2Q(p)}
\end{equation}

–¢—É—Ç $\lim\limits_{p\rightarrow \infty} e^{- 2Q(p)} = const > 0$, —Ç–æ–º—É –º–∞—î–º–æ, —â–æ
\begin{equation}
	K = -  \int\limits_\infty^1 \frac{e^{2Q(s)}}{s^2} ds =  \int\limits_1^\infty \frac{e^{2Q(s)}}{s^2} ds.
\end{equation}

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –æ—Ç—Ä–∏–º–∞–ª–∏ –Ω–æ–≤—É –≤–µ—Ä—Å—ñ—é $M(p)$:
\begin{equation}
	\label{eq:uniform_laplace_sol}
	M(p)= \left( \int\limits_p^1 \frac{e^{2Q(s)}}{s^2} ds + K \right) e^{- p - 2Q(p)} = 
	e^{- p - 2Q(p)} \int\limits_p^\infty \frac{e^{2Q(s)}}{s^2} ds
\end{equation}

\subsection{–ó–∞—Å—Ç–æ—Å—É–≤–∞–Ω–Ω—è —Ç–µ–æ—Ä–µ–º–∏ –¢–∞—É–±–µ—Ä–∞}

–î–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫–∏ $m(X)$ –Ω–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ, –∑–∞ —Ç–µ–æ—Ä–µ–º–æ—é –¢–∞—É–±–µ—Ä–∞ \eqref{eq:tauber_thm} –Ω–µ–æ–±—Ö—ñ–¥–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫—É $M(p)$ –ø—Ä–∏ $p \rightarrow 0$.

–Ø–∫—â–æ –∑–Ω–∞–π—Ç–∏ —Ç–∞–∫—ñ $C \in \RR$ —Ç–∞ $\alpha \in \RR^+$, —â–æ $M(p) \sim C \cdot p^{-\alpha}, \; p \rightarrow 0$, —Ç–æ –º–æ–∂–Ω–∞ —Å—Ç–≤–µ—Ä–¥–∂—É–≤–∞—Ç–∏, —â–æ $\int\limits_0^X m(x) dx \sim \frac{1}{\Gamma(\alpha + 1)} C X^\alpha, \; X \rightarrow \infty$. –í–∂–µ –∑–∞—Ä–∞–∑ –∑—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ $\alpha = 2$, –∞–¥–∂–µ —Ç–µ–æ—Ä–µ–º–∞ —Å–ø—Ä–∞–≤–µ–¥–ª–∏–≤–∞ –≤ –æ–±–∏–¥–≤–∞ –±–æ–∫–∏ —ñ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è \eqref{eq:bounds_inference}.

–î–ª—è —Ü—å–æ–≥–æ —Ä–æ–∑–≥–ª—è–Ω–µ–º—É –ø–æ–≤–µ–¥—ñ–Ω–∫—É –≤ –Ω—É–ª—ñ —Ç—Ä—å–æ—Ö –º–Ω–æ–∂–Ω–∏–∫—ñ–≤, –∑ —è–∫–∏—Ö —Å–∫–ª–∞–¥–∞—î—Ç—å—Å—è $M(p)$, –∞ —Å–∞–º–µ:
\begin{enumerate}
	\item $e^{-p}$;
	\item $e^{-2Q(p)}$;
	\item $\int\limits_p^\infty \frac{e^{2Q(s)}}{s^2} ds$,
\end{enumerate}

–©–æ–¥–æ –ø–µ—Ä—à–æ–≥–æ –º–Ω–æ–∂–Ω–∏–∫—É, —Ç–æ –≤ 0 –≤—ñ–Ω –ø—Ä—è–º—É—î –¥–æ 1, —Ç–æ–∂ –Ω—ñ –Ω–∞ —â–æ –Ω–µ –≤–ø–ª–∏–≤–∞—î, —ñ –π–æ–≥–æ –º–æ–∂–Ω–∞ –Ω–µ —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏.

–î–ª—è –Ω–∞—Å—Ç—É–ø–Ω–æ–≥–æ –∞–Ω–∞–ª—ñ–∑—É –¥–æ–≤–µ–¥–µ–º–æ –¥–µ—è–∫—ñ –ª–µ–º–∏.

\begin{lem}
	\label{eq:exp_q_p_asymptotics}
	$e^{-2Q(p)}$ –ø–æ–≤–æ–¥–∏—Ç—å—Å—è —è–∫ $p^{-2}$ –≤ 0, –∑ —Ç–æ—á–Ω—ñ—Å—Ç—é –¥–æ –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏, –∞ —Å–∞–º–µ:
	\begin{equation}
		\lim\limits_{p \rightarrow 0} \frac{e^{-2Q(p)}}{p^{-2}} = \exp\left(-2\int\limits_0^1 \frac{1-e^{-s}}{s} ds\right)
	\end{equation}
\end{lem}
\begin{proof}
	–î–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –ª—ñ–º—ñ—Ç—É –ø—Ä–æ–ª–æ–≥–∞—Ä–∏—Ñ–º—É—î–º–æ –≤–∏—Ä–∞–∑. –û—Ç—Ä–∏–º–∞—î–º–æ:
	\[
	\begin{split}
	2 \ln p - 2 Q(p) = 2 \ln p - 2 \int\limits_1^p \frac{e^{-s}}{s} ds &= 2 \int\limits_1^p \frac{1}{s} ds - 2 \int\limits_1^p \frac{e^{-s}}{s} ds = \\
		= 2  \int\limits_1^p \frac{1 - e^{-s}}{s} ds &= -2 \int\limits_p^1 \frac{1 - e^{-s}}{s} ds
	\end{split}
	\]
	–¢–µ–ø–µ—Ä, –ø—ñ–¥–≤—ñ–≤—à–∏ –¥–æ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏, –æ—Ç—Ä–∏–º–∞—î–º–æ:
	$$
		\frac{e^{-2Q(p)}}{p^{-2}} = \exp\left(-2 \int\limits_p^1 \frac{1 - e^{-s}}{s} ds\right)
	$$
	–Ø–∫—â–æ –¥–æ–≤–µ—Å—Ç–∏, —â–æ —ñ–Ω—Ç–µ–≥—Ä–∞–ª
	$$
		-2 \int\limits_0^1 \frac{1 - e^{-s}}{s} ds
	$$
	–∑–±—ñ–≥–∞—î—Ç—å—Å—è, —Ç–æ –ª–µ–º—É –±—É–¥–µ –¥–æ–≤–µ–¥–µ–Ω–æ, –∞–¥–∂–µ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–∞ ‚Äì –Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è, —ñ –º–æ–∂–Ω–∞ –ø–µ—Ä–µ—Ö–æ–¥–∏—Ç–∏ –¥–æ –ª—ñ–º—ñ—Ç—É –ø—ñ–¥ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–æ—é.
	–ó—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ
	$$
		-2 \int\limits_p^1 \frac{1 - e^{-s}}{s} ds
	$$
	–∑–±—ñ–≥–∞—î—Ç—å—Å—è –¥–ª—è $\forall p \in (0;~1]$. –î—ñ–π—Å–Ω–æ, –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è $ \frac{1 - e^{-s}}{s}$ –º–∞–∂–æ—Ä—É—î—Ç—å—Å—è $\frac{1}{s}$, —è–∫–∞, –≤ —Å–≤–æ—é —á–µ—Ä–≥—É, –º–∞—î —Å–∫—ñ–Ω—á–µ–Ω–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª—É:
	$$
		\int\limits_p^1 \frac{1}{s} ds = \ln 1 - \ln p = -\ln p,\quad p > 0
	$$
	–ù–µ–≤–∏–∑–Ω–∞—á–µ–Ω—ñ—Å—Ç—å –≤–∏–Ω–∏–∫–∞—î –ª–∏—à–µ –≤ —Ç–æ—á—Ü—ñ 0. –ó–Ω–∞–π–¥–µ–º–æ –ª—ñ–º—ñ—Ç –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó –≤ —Ç–æ—á—Ü—ñ 0:
	$$
		\lim\limits_{p\rightarrow 0} \frac{1 - e^{-s}}{s} = \<\text{–ø—Ä–∞–≤–∏–ª–æ –õ–æ–ø—ñ—Ç–∞–ª—è –¥–ª—è –Ω–µ–≤–∏–∑–Ω–∞—á–µ–Ω–Ω–æ—Å—Ç—ñ 0/0}\> = \lim\limits_{p\rightarrow 0} \frac{e^{-s}}{1} = 1
	$$
	–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –æ–±–º–µ–∂–µ–Ω–∞ –≤ –¥–µ—è–∫–æ–º—É $\varepsilon$-–æ–∫–æ–ª—ñ 0, —Ç–æ–º—É —ñ–Ω—Ç–µ–≥—Ä–∞–ª —Ç–∞–∫–æ–∂ –∑–±—ñ–∂–Ω–∏–π, —ñ –ª–µ–º—É –¥–æ–≤–µ–¥–µ–Ω–æ.
\end{proof}

\begin{lem}
	\label{eq:q_p_limited}
	–§—É–Ω–∫—Ü—ñ—è
	$$
		Q(p) = \int\limits_1^p \frac{e^{-s}}{s} ds
	$$
	  ‚Äì –æ–±–º–µ–∂–µ–Ω–∞ –Ω–∞ $[w; ~\infty], \; w>0$.
\end{lem}
\begin{proof}
	–ù–∞ –ø—Ä–æ–º—ñ–∂–∫—É $[1; ~\infty]$ –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –º–∞–∂–æ—Ä—É—î—Ç—å—Å—è —Ñ—É–Ω–∫—Ü—ñ—î—é $e^{-s}$, –∞ –Ω–∞ –ø—Ä–æ–º—ñ–∂–∫—É $[w; ~1]$ ‚Äì —Ñ—É–Ω–∫—Ü—ñ—î—é $\frac{1}{s}$, —Ç–æ–º—É, –∞–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥–æ–≤–µ–¥–µ–Ω–Ω—é –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ—ó –ª–µ–º–∏, —ñ–Ω—Ç–µ–≥—Ä–∞–ª –±—É–¥–µ –∑–±—ñ–∂–Ω–∏–π, —ñ:
	\begin{align*}
		Q(p) \leq \int\limits_1^\infty e^{-s} ds = \exp(-1) &,\quad p \geq 1 \\
		Q(p) \leq \int\limits_w^1 \frac{1}{s} ds = - \ln w &, \quad p \in [w;~1]
	\end{align*}
	–¢–∞–∫–∏–º —á–∏–Ω–æ–º, $Q(p) \leq \max\{-\ln w, \exp(-1)\}$.
\end{proof}

\begin{lem}
	–Ü–Ω—Ç–µ–≥—Ä–∞–ª
	$$
		\int\limits_0^\infty \frac{e^{2Q(s)}}{s^2} ds
	$$
	‚Äì –∑–±—ñ–∂–Ω–∏–π.
\end{lem}
\begin{proof}
	–°–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ –ª–µ–º—É \eqref{eq:exp_q_p_asymptotics}, –º–∞—î–º–æ, —â–æ –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –ø—Ä—è–º—É—î –¥–æ –¥–µ—è–∫–æ—ó –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ –ø—Ä–∏ $s \rightarrow 0$, –æ—Å–∫—ñ–ª—å–∫–∏ —î –æ–±–µ—Ä–Ω–µ–Ω–æ –ø—Ä–æ–ø–æ—Ä—Ü—ñ–π–Ω–æ—é –¥–æ —Ñ—É–Ω–∫—Ü—ñ—ó –∑ —Ç–æ—ó –ª–µ–º–∏. –¢–æ–º—É –≤ –¥–µ—è–∫–æ–º—É –ø—Ä–æ–∫–æ–ª–æ—Ç–æ–º—É $\varepsilon$-–æ–∫–æ–ª—ñ —Ç–æ—á–∫–∏ 0 –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –±—É–¥–µ –æ–±–º–µ–∂–µ–Ω–∞. –ù–∞ —ñ–Ω—Ç–µ—Ä–≤–∞–ª—ñ $[\varepsilon; ~\infty]$ –∑–∞ –ª–µ–º–æ—é \eqref{eq:q_p_limited}, $Q(s)$ ‚Äì –æ–±–º–µ–∂–µ–Ω–∞, –∞ —Ç–æ–º—É —ñ $\exp(2Q(s))$ —Ç–∞–∫–æ–∂. –¢–æ–º—É –∑–±—ñ–∂–Ω—ñ—Å—Ç—å –Ω–∞ —ñ–Ω—Ç–µ—Ä–≤–∞–ª—ñ $[\varepsilon; ~\infty]$ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è, —è–∫—â–æ –∑–±—ñ–≥–∞—î—Ç—å—Å—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª
	\[
		\int\limits_\varepsilon^\infty \frac{1}{s^2} ds.
	\]
	–ê –π–æ–≥–æ –∑–±—ñ–∂–Ω—ñ—Å—Ç—å ‚Äì –≤—ñ–¥–æ–º–∏–π —Ñ–∞–∫—Ç.
\end{proof}
		
–¢–∞–∫–∏–º —á–∏–Ω–æ–º, —Å–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ –¥–æ–≤–µ–¥–µ–Ω—ñ –ª–µ–º–∏, –º–∞—î–º–æ:
\begin{align*}
&M(p) \sim p^{-2} \cdot \exp\left(-2\int\limits_0^1 \frac{1-e^{-s}}{s} ds\right) \int\limits_0^\infty \frac{e^{2Q(s)}}{s^2} ds = \\
& = p^{-2} \cdot \exp\left(-2\int\limits_0^1 \frac{1-e^{-s}}{s} ds\right) \int\limits_0^\infty \exp\left(2\int\limits_1^s \frac{e^{-\tau}}{\tau} d\tau - 2 \ln s\right) ds = \\
& = p^{-2} \cdot \exp\left(-2\int\limits_0^1 \frac{1-e^{-s}}{s} ds\right) \int\limits_0^\infty \exp\left(2\int\limits_1^s \frac{e^{-\tau}}{\tau} d\tau - 2  \int\limits_1^s  \frac{1}{\tau} d\tau \right) ds = \\
& = p^{-2} \cdot \exp\left(-2\int\limits_0^1 \frac{1-e^{-s}}{s} ds\right) \int\limits_0^\infty \exp\left(-2\int\limits_1^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds = \\
& = p^{-2} \cdot \int\limits_0^\infty \exp\left( -2\int\limits_0^1 \frac{1-e^{-\tau}}{\tau} ds -2\int\limits_1^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds 
\end{align*}

–°–∫–ª–∞–≤—à–∏ —ñ–Ω—Ç–µ–≥—Ä–∞–ª–∏ –ø—ñ–¥ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–æ—é, –æ—Ç—Ä–∏–º–∞—î–º–æ:
\begin{equation}
	M(p) \sim p^{-2} \cdot \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds, \quad p \rightarrow 0.
\end{equation}

–¢–µ–ø–µ—Ä, –∑–∞ —Ç–µ–æ—Ä–µ–º–æ—é –¢–∞—É–±–µ—Ä–∞ –º–∞—î–º–æ –ø—Ä–∏ $X \rightarrow \infty$:
\begin{equation}
	\int\limits_0^X m(x) dx \sim \frac{1}{\Gamma(2 + 1)} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds \cdot X^2.
\end{equation}

–ê–±–æ, –ø—Ä–æ–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é–≤–∞–≤—à–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏,  –æ—Ç—Ä–∏–º–∞—î–º–æ:
\begin{equation}
	m(X) \sim \frac{2}{\Gamma(2 + 1)} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds \cdot X.
\end{equation}
\begin{equation}
	\label{eq:uniform_base_final}
	m(X) \sim  \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{1 - e^{-\tau}}{\tau} d\tau  \right) ds \cdot X, \quad X \rightarrow \infty.
\end{equation}

–¢—É—Ç –º–∏ –º–∞–ª–∏ –ø—Ä–∞–≤–æ –¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é–≤–∞—Ç–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏ –∑–∞ –ø—Ä–∞–≤–∏–ª–æ–º –õ–æ–ø—ñ—Ç–∞–ª—è, –∞–¥–∂–µ –º–∞—î –º—ñ—Å—Ü–µ –Ω–µ–≤–∏–∑–Ω–∞—á–µ–Ω—ñ—Å—Ç—å $\infty / \infty$.
