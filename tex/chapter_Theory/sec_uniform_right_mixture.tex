\subsection{–í–∏–≤–µ–¥–µ–Ω–Ω—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–æ–≥–æ —Ä—ñ–≤–Ω—è–Ω–Ω—è}

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ, —è–∫ —ñ –≤ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ–π —á–∞—Å—Ç–∏–Ω—ñ, –ø–æ—Ä—è–¥–æ–∫ –≤–∏–±–æ—Ä—É –≤—ñ–ª—å–Ω–∏—Ö –ø—Ä–æ–º—ñ–∂–∫—ñ–≤ –≤–æ–¥—ñ—è–º–∏ –Ω–µ –≤–ø–ª–∏–≤–∞—î –Ω–∞ —Ä–µ–∑—É–ª—å—Ç–∞—Ç, —Ç–æ–º—É –±—É–¥–µ–º–æ –≤–≤–∞–∂–∞—Ç–∏, —â–æ –ø—ñ—Å–ª—è –ø–∞—Ä–∫—É–≤–∞–Ω–Ω—è –æ–¥–Ω–æ–≥–æ –∞–≤—Ç–æ–º–æ–±—ñ–ª—è –ø–∞—Ä–∫–æ–≤–∫–∞ —Ä–æ–∑–±–∏–≤–∞—î—Ç—å—Å—è –Ω–∞ 2 —á–∞—Å—Ç–∏–Ω–∏, —ñ –ø—ñ—Å–ª—è —Ü—å–æ–≥–æ —Å–ø–æ—á–∞—Ç–∫—É –∑–∞–ø–æ–≤–Ω—é—î—Ç—å—Å—è –ª—ñ–≤–∞ —á–∞—Å—Ç–∏–Ω–∞, –∞ –ø–æ—Ç—ñ–º –ø—Ä–∞–≤–∞.

–ù–µ–æ–±—Ö—ñ–¥–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ $m(X)=\EE F(X)$. –ù–µ—Ö–∞–π $\xi \sim Uniform(0, X - 1)$ ‚Äì –≤–∏–ø–∞–¥–∫–æ–≤–∞ –≤–µ–ª–∏—á–∏–Ω–∞, —â–æ –≤–∏–∑–Ω–∞—á–∞—î –ø–æ–ª–æ–∂–µ–Ω–Ω—è –ª—ñ–≤–æ–≥–æ –∫—Ä–∞—é –ø–µ—Ä—à–æ–≥–æ –∞–≤—Ç–æ–º–æ–±—ñ–ª—è –Ω–∞ –ø–∞—Ä–∫–æ–≤—Ü—ñ —É –≤–∏–ø–∞–¥–∫—É –≤–∏–±–æ—Ä—É –º—ñ—Å—Ü—è –∑–∞ —Ä—ñ–≤–Ω–æ–º—ñ—Ä–Ω–∏–º —Ä–æ–∑–ø–æ–¥—ñ–ª–æ–º. –¢–æ–¥—ñ –º–∞—î–º–æ –Ω–∞—Å—Ç—É–ø–Ω—É —Ç–æ—Ç–æ–∂–Ω—ñ—Å—Ç—å:
\[
\begin{split}
	m(X) = \alpha(1+ m(X-1)) + \beta(1+ \EE(\EE(F(\xi) + F(X - 1 - \xi)  | \xi))) =\\
	=1 + \alpha m(X-1) + \beta \int\limits_0^{X-1} ùëö(ùë°) \frac{1}{ùëã ‚àí 1} ùëëùë° + \beta \int\limits_0^{X-1} ùëö(ùëã ‚àí ùë° ‚àí 1) \frac{1}{ùëã ‚àí 1} ùëëùë° 
\end{split}
\]

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –≤–∏–≤–µ–¥–µ–Ω–Ω—é —Ñ–æ—Ä–º—É–ª–∏ \eqref{eq:uniform_equation}, –æ—Ç—Ä–∏–º–∞—î–º–æ

	\label{eq:uniform_right_equation}
	m(X + 1) = 1 + \alpha m(X) + \frac{2\beta}{X} \int\limits_0^{X} ùëö(ùë°) dt,\quad \forall X > 0
\end{equation}

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥—ñ—è–º –≤ –≥–ª–∞–≤—ñ \ref{sec:uniform_base}, –º–∞—î–º–æ, —â–æ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è \eqref{eq:zero_state}, \eqref{eq:bounds} —Ç–∞ \eqref{eq:bounds_inference}.

\subsection{–ü–µ—Ä–µ—Ö—ñ–¥ –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞}
\label{sec:uniform_right_integral_laplace}

–°–ø—Ä–æ–±—É—î–º–æ —Ä–æ–∑–≤'—è–∑–∞—Ç–∏ \eqref{eq:uniform_right_equation} –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –ø–µ—Ä–µ—Ç–≤–æ—Ä–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞.

–û—Å–∫—ñ–ª—å–∫–∏ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è \eqref{eq:bounds}, —Ç–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –¥–ª—è $m(X)$ —ñ—Å–Ω—É—î. –î–æ —Ç–æ–≥–æ –∂, –∞–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥–æ –≤–∏–≤–µ–¥–µ–Ω–Ω—è –≤ –≥–ª–∞–≤—ñ \ref{sec:uniform_base} –º–∞—î–º–æ, —â–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ —ñ—Å–Ω—É—î —ñ –¥–ª—è —ñ–Ω—à–∏—Ö –¥–æ–¥–∞–Ω–∫—ñ–≤ –≤ –ø—Ä–∞–≤—ñ–π —á–∞—Å—Ç–∏–Ω—ñ —Ä—ñ–≤–Ω—è–Ω–Ω—è \eqref{eq:uniform_right_equation}.

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –æ—Ç—Ä–∏–º–∞–ª–∏ —ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è –≤ —Ç–µ—Ä–º—ñ–Ω–∞—Ö –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞, —è–∫–µ –≤–∂–µ –º–æ–∂–Ω–∞ —Ä–æ–∑–≤'—è–∑–∞—Ç–∏, –∞–¥–∂–µ –Ω–µ–º–∞ –∑—Å—É–≤—É:
\begin{equation}
	\label{eq:uniform_right_laplace_integral}
	e^p M(p) = \alpha M(p) + 2\beta \int\limits_p^\infty \frac{M(s)}{s} ds + \frac{1}{p}
\end{equation}

–ü—Ä–æ–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é—î–º–æ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏ —Ä—ñ–≤–Ω—è–Ω–Ω—è –∑–∞ $p$:
\begin{equation}
	e^p M(p) + e^p \dot M(p) =\alpha \dot M(p) - 2 \beta \frac{M(p)}{p}  - \frac1{p^2}
\end{equation}

–í–∏—Ä–∞–∑–∏–º–æ $\dot M(p)$ –∑ —Ü—å–æ–≥–æ —Ä—ñ–≤–Ω—è–Ω–Ω—è:
\begin{equation}
	\label{eq:uniform_right_laplace_diff}
	\dot M(p) = - M(p)\left(\frac{e^p}{e^p - \alpha} + \frac{2\beta}{p(e^p - \alpha)}\right) - \frac{1}{p^2(e^p - \alpha)}
\end{equation}

–†–æ–∑–≤'—è–∂–µ–º–æ –æ—Ç—Ä–∏–º–∞–Ω–µ –¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ–π–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è. –°–ø–æ—á–∞—Ç–∫—É —Ä–æ–∑–≤'—è–∂–µ–º–æ –æ–¥–Ω–æ—Ä—ñ–¥–Ω—É —á–∞—Å—Ç–∏–Ω—É:
\begin{align*}
	\dot M_h (p) &= - M_h (p)\left(\frac{e^p}{e^p - \alpha} + \frac{2\beta}{p(e^p - \alpha)}\right) \\
	\frac{\dot M_h (p)}{M_h (p)} &= -\left(\frac{e^p}{e^p - \alpha} + \frac{2\beta}{p(e^p - \alpha)}\right) \\
	\int\limits_1^p \frac{\dot M_h (s)}{M_h (s)} ds &= -\int\limits_1^p \left(\frac{e^s}{e^s - \alpha} + \frac{2\beta}{s(e^s - \alpha)}\right) ds \\
	\left. \ln{M_h (s)}\right|_1^p &= - \int\limits_1^p \frac{e^s}{e^s - \alpha} ds  - 2 \int\limits_1^p \frac{\beta}{s(e^s - \alpha)} ds
\end{align*}

–ü–æ–∑–Ω–∞—á–∏–º–æ
\begin{equation}
	\label{eq:almost_li_alpha}
	Q_\alpha(p) := \int\limits_1^p \frac{\beta}{s(e^s - \alpha)} ds
\end{equation}

–û—Å–∫—ñ–ª—å–∫–∏
\begin{align*}
	\int\limits_1^p \frac{e^s}{e^s - \alpha} ds &= \< u = e^s - \alpha,~ du = e^s ds = (u+\alpha) ds \> = \\
	&= \int\limits_{e-\alpha}^{e^p -\alpha} \frac{u + \alpha}{u} (u+\alpha)^{-1} du = 
	\int\limits_{e-\alpha}^{e^p -\alpha} \frac{du}{u} = \\
	&= \log(e^p - \alpha) - \log(e-\alpha) = \log \frac{e^p - \alpha}{e - \alpha} \text{,}
\end{align*}

—Ç–æ
\begin{align*}
	\ln{M_h (p)} &= \ln{M_h (1)} - \log \frac{e^p - \alpha}{e - \alpha} - 2Q_\alpha(p) \\
	M_h (p) &= M_h (1) \cdot \frac{e - \alpha}{e^p - \alpha} \cdot e^{- 2Q_\alpha(p)} \cdot const
\end{align*}

–û—Å–∫—ñ–ª—å–∫–∏ $M(1)$ —Ç–∞ $(e-\alpha)$ –º–æ–∂–Ω–∞ –≤–∫–ª—é—á–∏—Ç–∏ –≤ –∫–æ–Ω—Å—Ç–∞–Ω—Ç—É, —Ç–æ –º–∞—î–º–æ —Ä–æ–∑–≤'—è–∑–æ–∫
\begin{equation}
	\label{eq:uniform_right_laplace_homogen_sol}
	M_h (p) = C \cdot \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1}, \quad \forall C \in \RR
\end{equation}

–î—ñ–π—Å–Ω–æ, –ø–µ—Ä–µ–≤—ñ—Ä–∏–º–æ —Ü–µ–π —Ä–æ–∑–≤'—è–∑–æ–∫:
\[
\begin{split}
	\dot M_h (p) &= C \cdot \left(\frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}\right)'=\\
	 &= - C \cdot \left(\frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}\right)^2 \cdot \left(e^p  + (e^p - \alpha) \frac{2\beta}{p(e^p - \alpha)}   \right) e^{2Q_\alpha(p)} =\\
	 &= -C \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1} \cdot \left(\frac{e^p}{e^p - \alpha}  + \frac{2\beta}{p(e^p - \alpha)}   \right) =\\
	 &= - M(p) \cdot \left(\frac{e^p}{e^p - \alpha}  + \frac{2\beta}{p(e^p - \alpha)}   \right)
\end{split}
\]

–ù–µ—Å–∫–ª–∞–¥–Ω–æ –ø–æ–º—ñ—Ç–∏—Ç–∏, —â–æ –æ—Ç—Ä–∏–º–∞–ª–∏ –≤–∏—Ö—ñ–¥–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è. –¢–µ–ø–µ—Ä –∑–∞—Å—Ç–æ—Å—É—î–º–æ –º–µ—Ç–æ–¥ –≤–∞—Ä—ñ–∞—Ü—ñ—ó –¥–æ–≤—ñ–ª—å–Ω–∏—Ö —Å—Ç–∞–ª–∏—Ö:
$$
	M(p) = C(p) \cdot \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1} 
$$

–ü—Ä–æ–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é–≤–∞–≤—à–∏ –∑–∞ $p$, –æ—Ç—Ä–∏–º–∞—î–º–æ:
\begin{align*}
	\dot M(p) &= \dot C(p) \cdot \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1} -C(p) \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1} \cdot \\
	&\cdot \left(\frac{e^p}{e^p - \alpha}  + \frac{2\beta}{p(e^p - \alpha)}   \right) 
\end{align*}

–ó —ñ–Ω—à–æ—ó —Å—Ç–æ—Ä–æ–Ω–∏, –∑ \eqref{eq:uniform_right_laplace_diff} –º–∞—î–º–æ
$$
	 \dot M(p) = - C(p) \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1}  \left(\frac{e^p}{e^p - \alpha} + \frac{2\beta}{p(e^p - \alpha)}\right) - \frac{1}{p^2(e^p - \alpha)} \quad	
$$
–¢–æ–º—É
$$
	\dot C(p) = - \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right) \cdot  \frac{1}{p^2(e^p - \alpha)} =  - \frac{e^{2Q_\alpha(p)}}{p^2}
$$

–¢–æ–¥—ñ –ø—Ä–æ—Å—Ç–∏–º —ñ–Ω—Ç–µ–≥—Ä—É–≤–∞–Ω–Ω—è–º –≤ –º–µ–∂–∞—Ö –≤—ñ–¥ 1 –¥–æ $p$ –æ—Ç—Ä–∏–º—É—î–º–æ:
\begin{equation}
	C(p) = -  \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + const
\end{equation}

–Ü —Ç–æ–¥—ñ –æ—Ç—Ä–∏–º—É—î–º–æ –≤–∏—Ä–∞–∑ –¥–ª—è $M(p)$:
\begin{equation}
\begin{split}
	M(p)=- \left( \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + const \right) \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1}  =\\
	= - \left( \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}, \quad K \in \RR
\end{split}
\end{equation}

–ü–µ—Ä–µ–≤—ñ—Ä–∏–º–æ –æ—Ç—Ä–∏–º–∞–Ω–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:
\begin{equation*}
\begin{split}
	&\dot M(p) = -\left( \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right)' \frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}} -\left( \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right)\cdot \\ 
	&\cdot \left(\frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}\right)' = -\frac{e^{2Q_\alpha(p)}}{p^2} \frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}  + \left( \int\limits_1^p \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \cdot \\
	& \cdot \left((e^p - \alpha)  e^{2Q_\alpha(p)}\right)^{-1} \cdot \left(\frac{e^p}{e^p - \alpha}  + \frac{2\beta}{p(e^p - \alpha)}   \right)= -\frac{1}{p^2(e^p - \alpha)} -\\
	&- M(p)\left(\frac{e^p}{e^p - \alpha}  + \frac{2\beta}{p(e^p - \alpha)}   \right)
\end{split}
\end{equation*}

–ü–µ—Ä–µ–≤—ñ—Ä–µ–Ω–æ. –¢–æ–¥—ñ –æ—Å—Ç–∞—Ç–æ—á–Ω–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç –±–µ–∑ –≤–∏—Ä–∞—Ö—É–≤–∞–Ω–Ω—è –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏:
\begin{equation}
	\label{eq:uniform_right_laplace_sol_initial}
	M_\alpha(p)= \left( \int\limits_p^1 \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}
\end{equation}

–ó–∞–∑–Ω–∞—á–∏–º–æ, —â–æ —É –≤–∏–ø–∞–¥–∫—É $\alpha=0$ –º–∏ –º–∞—î–º–æ –≤–∏–ø–∞–¥–æ–∫ –∑ –≥–ª–∞–≤–∏ \ref{sec:uniform_base}, —Ç–æ–∂ —ñ —Ñ–æ—Ä–º—É–ª–∞ \eqref{eq:uniform_right_laplace_sol_initial} –º–∞—î —Å–ø—ñ–≤–ø–∞–¥–∞—Ç–∏ –∑ \eqref{eq:uniform_laplace_sol_initial} –ø—Ä–∏ $\alpha = 0$. –°–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ —Ç–µ, —â–æ $Q_0(p) \equiv Q(p)$, –æ—Ç—Ä–∏–º—É—î–º–æ, —â–æ $M_0(p) \equiv M(p) \; \forall p > 0$.

\subsection{–í–∏–∑–Ω–∞—á–µ–Ω–Ω—è –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ —É —Ä–æ–∑–≤'—è–∑–∫—É}

–£ \ref{sec:uniform_integral_laplace} –±—É–ª–æ –¥–æ–≤–µ–¥–µ–Ω–æ, —â–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ —ñ—Å–Ω—É—î –Ω–µ —Ç—ñ–ª—å–∫–∏ –¥–ª—è $m(X)$, –∞ —ñ –¥–ª—è $m(X+1)$, –¥–æ —Ç–æ–≥–æ –∂,
\begin{equation}
	\Lapl{m(X+1)} = e^p M(p).
\end{equation}

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥–æ–≤–æ–¥–∏—Ç—å—Å—è —Ç–æ–π —Å–∞–º–∏–π —Ñ–∞–∫—Ç, –∞–ª–µ –¥–ª—è –≤–∏–ø–∞–¥–∫—É —Å—É–º—ñ—à—ñ —Ä—ñ–≤–Ω–æ–º—ñ—Ä–Ω–æ–≥–æ —Ä–æ–∑–ø–æ–¥—ñ–ª—É —Ç–∞ —Ä–æ–∑–ø–æ–¥—ñ–ª—É –ë–µ—Ä–Ω—É–ª—ñ. –¢–∞–∫–∏–º —á–∏–Ω–æ–º, –º–∞—î–º–æ
\begin{equation}
	\Lapl{m_\alpha(X+1)} =\tilde M_\alpha(p)= \left( \int\limits_p^1 \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \frac{e^p}{e^p - \alpha} e^{- 2Q_\alpha(p)}.
\end{equation}

–û—Å–∫—ñ–ª—å–∫–∏ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ ‚Äì –∞–Ω–∞–ª—ñ—Ç–∏—á–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –≤ –¥–µ—è–∫—ñ–π –ø—Ä–∞–≤—ñ–π –ø—ñ–≤–ø–ª–æ—â–∏–Ω—ñ –∫–æ–º–ø–ª–µ–∫—Å–Ω–æ–≥–æ –ø—Ä–æ—Å—Ç–æ—Ä—É, —Ç–æ $\tilde M_\alpha(p) \rightarrow 0,\; p \rightarrow +\infty$.

–†–æ–∑–≥–ª—è–Ω–µ–º–æ $Q_\alpha(p)$ ($p$ —Ä–æ–∑–≥–ª—è–¥–∞—î–º–æ –Ω–∞ –¥—ñ–π—Å–Ω—ñ–π –≤—ñ—Å—ñ):
\begin{equation}
\begin{split}
	\label{eq:q_alpha_p_major}
	&Q_\alpha(p) = \int\limits_1^p \frac{1-\alpha}{s(e^s - \alpha)} ds <  \int\limits_1^\infty \frac{1 - \alpha}{s(e^s - \alpha)} ds < 
	\int\limits_1^\infty \frac{1- \alpha}{e^s - \alpha} ds<\\
	&< \int\limits_1^\infty \frac{1}{e^s} ds = \exp(-1) - \exp(-\infty) = \exp(-1)
\end{split}
\end{equation}

–û—Å—Ç–∞–Ω–Ω—ñ–π –ø–µ—Ä–µ—Ö—ñ–¥ –Ω–µ—Ä—ñ–≤–Ω–æ—Å—Ç—ñ –ø–æ—è—Å–Ω—é—î—Ç—å—Å—è –¥–æ—Å–∏—Ç—å –ø—Ä–æ—Å—Ç–æ:
$$
\frac{1-\alpha}{u - \alpha} < \frac{1}{\alpha}, \; u > 1 \Leftrightarrow u - u \alpha = u(1-\alpha) < u - \alpha, \; u > 1
$$

–¢–æ–±—Ç–æ $Q_\alpha(p)$ - –æ–±–º–µ–∂–µ–Ω–∞ –Ω–∞ $[1; \infty]$. –¢–æ–º—É –æ–±–º–µ–∂–µ–Ω–∏–º–∏ –Ω–∞ —Ü—ñ–π –≤—ñ—Å—ñ –±—É–¥—É—Ç—å —ñ $e^{\pm 2Q_\alpha(p)}$. –¢–∞–∫–æ–∂ –∑—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ —è–∫—â–æ —ñ–Ω—Ç–µ–≥—Ä—É–≤–∞—Ç–∏ –ø–æ –¥—ñ–π—Å–Ω—ñ–π –≤—ñ—Å—ñ, —Ç–æ $Q_\alpha(p)$ ‚Äì –º–æ–Ω–æ—Ç–æ–Ω–Ω–æ –∑—Ä–æ—Å—Ç–∞—é—á–∞ –∑–∞ $p$. –¢–æ–º—É
\begin{equation}
	0 = \tilde M_\alpha(\infty) = \lim_{p\rightarrow \infty} \tilde M_\alpha(p) = \left( \int\limits_\infty^1 \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \lim_{p\rightarrow \infty} e^{- 2Q_\alpha(p)}
\end{equation}

–¢—É—Ç $\lim\limits_{p\rightarrow \infty} e^{- 2Q_\alpha(p)} = const > 0$, —Ç–æ–º—É –º–∞—î–º–æ, —â–æ
\begin{equation}
	K = -  \int\limits_\infty^1 \frac{e^{2Q_\alpha(s)}}{s^2} ds =  \int\limits_1^\infty \frac{e^{2Q_\alpha(s)}}{s^2} ds.
\end{equation}

–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –æ—Ç—Ä–∏–º–∞–ª–∏ –Ω–æ–≤—É –≤–µ—Ä—Å—ñ—é $M(p)$:
\begin{equation}
\begin{split}
	\label{eq:uniform_right_laplace_sol}
	M_\alpha(p)&= \left( \int\limits_p^1 \frac{e^{2Q_\alpha(s)}}{s^2} ds + K \right) \frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}}
 = \\
	&=\frac{1}{(e^p - \alpha)  e^{2Q_\alpha(p)}} \int\limits_p^\infty \frac{e^{2Q_\alpha(s)}}{s^2} ds
\end{split}
\end{equation}

\subsection{–ó–∞—Å—Ç–æ—Å—É–≤–∞–Ω–Ω—è —Ç–µ–æ—Ä–µ–º–∏ –¢–∞—É–±–µ—Ä–∞}

–î–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫–∏ $m_\alpha(X)$ –Ω–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ, –∑–∞ —Ç–µ–æ—Ä–µ–º–æ—é –¢–∞—É–±–µ—Ä–∞ \eqref{eq:tauber_thm} –Ω–µ–æ–±—Ö—ñ–¥–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –∞—Å–∏–º–ø—Ç–æ—Ç–∏–∫—É $M_\alpha(p)$ –ø—Ä–∏ $p \rightarrow 0$.

–Ø–∫—â–æ –∑–Ω–∞–π—Ç–∏ —Ç–∞–∫—ñ $C \in \RR$ —Ç–∞ $\delta \in \RR^+$, —â–æ $M_\alpha(p) \sim C \cdot p^{-\delta}, \; p \rightarrow 0$, —Ç–æ –º–æ–∂–Ω–∞ —Å—Ç–≤–µ—Ä–¥–∂—É–≤–∞—Ç–∏, —â–æ $\int\limits_0^X m_\alpha(x) dx \sim \frac{1}{\Gamma(\delta + 1)} C X^\delta, \; X \rightarrow \infty$. –í–∂–µ –∑–∞—Ä–∞–∑ –∑—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ $\delta = 2$, –∞–¥–∂–µ —Ç–µ–æ—Ä–µ–º–∞ —Å–ø—Ä–∞–≤–µ–¥–ª–∏–≤–∞ –≤ –æ–±–∏–¥–≤–∞ –±–æ–∫–∏ —ñ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è \eqref{eq:bounds_inference}.

–î–ª—è —Ü—å–æ–≥–æ —Ä–æ–∑–≥–ª—è–Ω–µ–º—É –ø–æ–≤–µ–¥—ñ–Ω–∫—É –≤ –Ω—É–ª—ñ —Ç—Ä—å–æ—Ö –º–Ω–æ–∂–Ω–∏–∫—ñ–≤, –∑ —è–∫–∏—Ö —Å–∫–ª–∞–¥–∞—î—Ç—å—Å—è $M_\alpha(p)$, –∞ —Å–∞–º–µ:
\begin{enumerate}
	\item $\frac{1}{e^p - \alpha}$;
	\item $e^{-2Q_\alpha(p)}$;
	\item $\int\limits_p^\infty \frac{e^{2Q_\alpha(s)}}{s^2} ds$,
\end{enumerate}

–©–æ–¥–æ –ø–µ—Ä—à–æ–≥–æ –º–Ω–æ–∂–Ω–∏–∫—É, —Ç–æ –≤ 0 –≤—ñ–Ω, –æ—á–µ–≤–∏–¥–Ω–æ, –ø—Ä—è–º—É—î –¥–æ $\frac{1}{1-\alpha}$,

–î–ª—è –Ω–∞—Å—Ç—É–ø–Ω–æ–≥–æ –∞–Ω–∞–ª—ñ–∑—É –¥–æ–≤–µ–¥–µ–º–æ –¥–µ—è–∫—ñ –ª–µ–º–∏.

\begin{lem}
	\label{eq:exp_q_alpha_p_asymptotics}
	$e^{-2Q(p)}$ –ø–æ–≤–æ–¥–∏—Ç—å—Å—è —è–∫ $p^{-2}$ –≤ 0, –∑ —Ç–æ—á–Ω—ñ—Å—Ç—é –¥–æ –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏, –∞ —Å–∞–º–µ:
	\begin{equation}
		\lim\limits_{p \rightarrow 0} \frac{e^{-2Q(p)}}{p^{-2}} = \exp\left(-2\int\limits_0^1 \frac{e^s  - 1}{s(s^s - \alpha)} ds\right)
	\end{equation}
\end{lem}
\begin{proof}
	–î–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –ª—ñ–º—ñ—Ç—É –ø—Ä–æ–ª–æ–≥–∞—Ä–∏—Ñ–º—É—î–º–æ –≤–∏—Ä–∞–∑. –û—Ç—Ä–∏–º–∞—î–º–æ:
	\[
	\begin{split}
	2 \ln p - 2 Q_\alpha(p) = 2 \ln p &- 2 \int\limits_1^p \frac{1 - \alpha}{s(e^s - \alpha)} ds = 2 \int\limits_1^p \frac{1}{s} ds - \\
	- 2 \int\limits_1^p \frac{1 - \alpha}{s(e^s - \alpha)} ds &= 2  \int\limits_1^p \frac{e^s - 1}{s(e^s - \alpha)} ds = -2 \int\limits_p^1 \frac{e^s - 1}{s(e^s - \alpha)} ds
	\end{split}
	\]
	–¢–µ–ø–µ—Ä, –ø—ñ–¥–≤—ñ–≤—à–∏ –¥–æ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏, –æ—Ç—Ä–∏–º–∞—î–º–æ:
	$$
		\frac{e^{-2Q_\alpha(p)}}{p^{-2}} = \exp\left(-2 \int\limits_p^1 \frac{e^s - 1}{s(e^s - \alpha)} ds\right)
	$$
	–Ø–∫—â–æ –¥–æ–≤–µ—Å—Ç–∏, —â–æ —ñ–Ω—Ç–µ–≥—Ä–∞–ª
	$$
		-2 \int\limits_0^1 \frac{e^s - 1}{s(e^s - \alpha)} ds
	$$
	–∑–±—ñ–≥–∞—î—Ç—å—Å—è, —Ç–æ –ª–µ–º—É –±—É–¥–µ –¥–æ–≤–µ–¥–µ–Ω–æ, –∞–¥–∂–µ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–∞ ‚Äì –Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è, —ñ –º–æ–∂–Ω–∞ –ø–µ—Ä–µ—Ö–æ–¥–∏—Ç–∏ –¥–æ –ª—ñ–º—ñ—Ç—É –ø—ñ–¥ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–æ—é.
	–ó—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ
	$$
		-2 \int\limits_p^1 \frac{e^s - 1}{s(e^s - \alpha)} ds
	$$
	–∑–±—ñ–≥–∞—î—Ç—å—Å—è –¥–ª—è $\forall p \in (0;~1]$. –î—ñ–π—Å–Ω–æ, –æ—Å–∫—ñ–ª—å–∫–∏ $e^s - 1 < e^s - \alpha$, –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è $ \frac{1 - e^{-s}}{s}$ –º–∞–∂–æ—Ä—É—î—Ç—å—Å—è $\frac{1}{s}$, —è–∫–∞, –≤ —Å–≤–æ—é —á–µ—Ä–≥—É, –º–∞—î —Å–∫—ñ–Ω—á–µ–Ω–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª—É:
	$$
		\int\limits_p^1 \frac{1}{s} ds = \ln 1 - \ln p = -\ln p,\quad p > 0
	$$
	–ù–µ–≤–∏–∑–Ω–∞—á–µ–Ω—ñ—Å—Ç—å –≤–∏–Ω–∏–∫–∞—î –ª–∏—à–µ –≤ —Ç–æ—á—Ü—ñ 0. –ó–Ω–∞–π–¥–µ–º–æ –ª—ñ–º—ñ—Ç –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó –≤ —Ç–æ—á—Ü—ñ 0:
\begin{align*}
	\lim\limits_{p\rightarrow 0} \frac{e^s - 1}{s(e^s - \alpha)} = \<\text{–ø—Ä–∞–≤–∏–ª–æ –õ–æ–ø—ñ—Ç–∞–ª—è –¥–ª—è –Ω–µ–≤–∏–∑–Ω–∞—á–µ–Ω–Ω–æ—Å—Ç—ñ 0/0}\> = \\
	= \lim\limits_{p\rightarrow 0} \frac{e^s}{s e^s + (e^s-\alpha)} = \frac{1}{1-\alpha}
\end{align*}
	–¢–∞–∫–∏–º —á–∏–Ω–æ–º, –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –æ–±–º–µ–∂–µ–Ω–∞ –≤ –¥–µ—è–∫–æ–º—É $\varepsilon$-–æ–∫–æ–ª—ñ 0, —Ç–æ–º—É —ñ–Ω—Ç–µ–≥—Ä–∞–ª —Ç–∞–∫–æ–∂ –∑–±—ñ–∂–Ω–∏–π, —ñ –ª–µ–º—É –¥–æ–≤–µ–¥–µ–Ω–æ.
\end{proof}

\begin{lem}
	\label{eq:q_alpha_p_limited}
	–§—É–Ω–∫—Ü—ñ—è
	$$
		Q_\alpha(p) = \int\limits_1^p \frac{1- \alpha}{s(e^s - \alpha)} ds
	$$
	  ‚Äì –æ–±–º–µ–∂–µ–Ω–∞ –Ω–∞ $[w; ~\infty], \; w>0$.
\end{lem}
\begin{proof}
	–ù–∞ –ø—Ä–æ–º—ñ–∂–∫—É $[1; ~\infty]$ –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –º–∞–∂–æ—Ä—É—î—Ç—å—Å—è —Ñ—É–Ω–∫—Ü—ñ—î—é $e^{-s}$ (–¥–∏–≤. \eqref{eq:q_alpha_p_major}), –∞ –Ω–∞ –ø—Ä–æ–º—ñ–∂–∫—É $[w; ~1]$ ‚Äì —Ñ—É–Ω–∫—Ü—ñ—î—é $\frac{1}{s}$, –∞–¥–∂–µ $1- \alpha < e^s - \alpha, ~ s > 0$. –¢–æ–º—É, –∞–Ω–∞–ª–æ–≥—ñ—á–Ω–æ –¥–æ–≤–µ–¥–µ–Ω–Ω—é –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ –ª–µ–º–∏, —ñ–Ω—Ç–µ–≥—Ä–∞–ª –±—É–¥–µ –∑–±—ñ–∂–Ω–∏–π, —ñ:
	\begin{align*}
		Q_\alpha(p) \leq \int\limits_1^\infty e^{-s} ds = \exp(-1) &,\quad p \geq 1 \\
		Q_\alpha(p) \leq \int\limits_w^1 \frac{1}{s} ds = - \ln w &, \quad p \in [w;~1]
	\end{align*}
	–¢–∞–∫–∏–º —á–∏–Ω–æ–º, $Q_\alpha(p) \leq \max\{-\ln w, \exp(-1)\}$.
\end{proof}

\begin{lem}
	–Ü–Ω—Ç–µ–≥—Ä–∞–ª
	$$
		\int\limits_0^\infty \frac{e^{2Q_\alpha(s)}}{s^2} ds
	$$
	‚Äì –∑–±—ñ–∂–Ω–∏–π.
\end{lem}
\begin{proof}
	–°–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ –ª–µ–º—É \eqref{eq:exp_q_alpha_p_asymptotics}, –º–∞—î–º–æ, —â–æ –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –ø—Ä—è–º—É—î –¥–æ –¥–µ—è–∫–æ—ó –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ –ø—Ä–∏ $s \rightarrow 0$, –æ—Å–∫—ñ–ª—å–∫–∏ —î –æ–±–µ—Ä–Ω–µ–Ω–æ –ø—Ä–æ–ø–æ—Ä—Ü—ñ–π–Ω–æ—é –¥–æ —Ñ—É–Ω–∫—Ü—ñ—ó –∑ —Ç–æ—ó –ª–µ–º–∏. –¢–æ–º—É –≤ –¥–µ—è–∫–æ–º—É –ø—Ä–æ–∫–æ–ª–æ—Ç–æ–º—É $\varepsilon$-–æ–∫–æ–ª—ñ —Ç–æ—á–∫–∏ 0 –ø—ñ–¥—ñ–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –±—É–¥–µ –æ–±–º–µ–∂–µ–Ω–∞. –ù–∞ —ñ–Ω—Ç–µ—Ä–≤–∞–ª—ñ $[\varepsilon; ~\infty]$ –∑–∞ –ª–µ–º–æ—é \eqref{eq:q_alpha_p_limited}, $Q_\alpha(s)$ ‚Äì –æ–±–º–µ–∂–µ–Ω–∞, –∞ —Ç–æ–º—É —ñ $\exp(2Q_\alpha(s))$ —Ç–∞–∫–æ–∂. –¢–æ–º—É –∑–±—ñ–∂–Ω—ñ—Å—Ç—å –Ω–∞ —ñ–Ω—Ç–µ—Ä–≤–∞–ª—ñ $[\varepsilon; ~\infty]$ –≤–∏–∫–æ–Ω—É—î—Ç—å—Å—è, —è–∫—â–æ –∑–±—ñ–≥–∞—î—Ç—å—Å—è —ñ–Ω—Ç–µ–≥—Ä–∞–ª
	\[
		\int\limits_\varepsilon^\infty \frac{1}{s^2} ds.
	\]
	–ê –π–æ–≥–æ –∑–±—ñ–∂–Ω—ñ—Å—Ç—å ‚Äì –≤—ñ–¥–æ–º–∏–π —Ñ–∞–∫—Ç.
\end{proof}
		
–¢–∞–∫–∏–º —á–∏–Ω–æ–º, —Å–ø–∏—Ä–∞—é—á–∏—Å—å –Ω–∞ –¥–æ–≤–µ–¥–µ–Ω—ñ –ª–µ–º–∏, –º–∞—î–º–æ –ø—Ä–∏ $p \rightarrow 0$:
\begin{align*}
&M_\alpha(p) \sim \frac{p^{-2}}{1-\alpha} \cdot \exp\left(-2\int\limits_0^1 \frac{e^s  - 1}{s(e^s - \alpha)} ds\right) \int\limits_0^\infty \frac{e^{2Q_\alpha(s)}}{s^2} ds = \\
& = \frac{p^{-2}}{1-\alpha} \cdot \exp\left(-2\int\limits_0^1 \frac{e^s  - 1}{s(e^s - \alpha)} ds\right) \int\limits_0^\infty \exp\left(2\int\limits_1^s \frac{1- \alpha}{\tau(e^\tau - \alpha)} d\tau - 2 \ln s\right) ds = \\
& = \frac{p^{-2}}{1-\alpha} \cdot \exp\left(-2\int\limits_0^1 \frac{e^s  - 1}{s(e^s - \alpha)} ds\right) \int\limits_0^\infty \exp\left(2\int\limits_1^s \frac{1- \alpha}{\tau(e^\tau - \alpha)} d\tau - 2  \int\limits_1^s  \frac{1}{\tau} d\tau \right) ds = \\
& = \frac{p^{-2}}{1-\alpha} \cdot \exp\left(-2\int\limits_0^1 \frac{e^s  - 1}{s(e^s - \alpha)} ds\right) \int\limits_0^\infty \exp\left(-2\int\limits_1^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds = \\
& = \frac{p^{-2}}{1-\alpha} \cdot \int\limits_0^\infty \exp\left( -2\int\limits_0^1 \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau -2\int\limits_1^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds 
\end{align*}

–°–∫–ª–∞–≤—à–∏ —ñ–Ω—Ç–µ–≥—Ä–∞–ª–∏ –ø—ñ–¥ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ç–æ—é, –æ—Ç—Ä–∏–º–∞—î–º–æ:
\begin{equation}
	M_\alpha(p) \sim p^{-2} \cdot \frac{1}{1-\alpha} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds, \quad p \rightarrow 0.
\end{equation}

–¢–µ–ø–µ—Ä, –∑–∞ —Ç–µ–æ—Ä–µ–º–æ—é –¢–∞—É–±–µ—Ä–∞ –º–∞—î–º–æ –ø—Ä–∏ $X \rightarrow \infty$:
\begin{equation}
	\int\limits_0^X m_\alpha(x) dx \sim \frac{1}{(1-\alpha)\Gamma(2 + 1)} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds \cdot X^2.
\end{equation}

–ê–±–æ, –ø—Ä–æ–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é–≤–∞–≤—à–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏,  –æ—Ç—Ä–∏–º–∞—î–º–æ:
\begin{equation}
	m_\alpha(X) \sim \frac{2}{(1- \alpha) \Gamma(2 + 1)} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds \cdot X.
\end{equation}
\begin{equation}
	\label{eq:uniform_right_final}
	m_\alpha(X) \sim \frac{1}{1-\alpha} \int\limits_0^\infty \exp\left( -2\int\limits_0^s \frac{e^{\tau} - 1}{\tau(e^\tau - \alpha)} d\tau  \right) ds \cdot X, \quad X \rightarrow \infty.
\end{equation}

–¢—É—Ç –º–∏ –º–∞–ª–∏ –ø—Ä–∞–≤–æ –¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ—é–≤–∞—Ç–∏ –æ–±–∏–¥–≤—ñ —á–∞—Å—Ç–∏–Ω–∏ –∑–∞ –ø—Ä–∞–≤–∏–ª–æ–º –õ–æ–ø—ñ—Ç–∞–ª—è, –∞–¥–∂–µ –º–∞—î –º—ñ—Å—Ü–µ –Ω–µ–≤–∏–∑–Ω–∞—á–µ–Ω—ñ—Å—Ç—å $\infty / \infty$.
